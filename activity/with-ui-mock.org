{^"Next step"\n  Do you want to continue? \n[Yes]|[No]\n}
!endprocedure

!procedure _choose()
skinparam handwritten true\n{+\n<b>an example\nchoose one option\n()one\n()two\n[ok]\n}
!endprocedure

start

if (\n{{\nsalt\n{^"Next step"\n  Do you want to continue? \n[Yes]|[No]\n}\n}}\n) then (yes)

:SALT(choose);

else (no)

:Bye;

endif
